\section*{Problem}
Let $M_2(\mathbb{Z})$ be a ring of $2\times2$ matrices with integer entries, and let $A\in M_2(\mathbb{Z})$.\\
Prove that 
\begin{itemize}
    \item $A \text{ is a unit in } M_2(\mathbb{Z}) \Longleftrightarrow  \det(A) = 1 \text{ or } -1$
\end{itemize}

\section*{Proof}

\subsection*{$(\Leftarrow)$}
Suppose $A = \begin{bmatrix} a & b \\ c & d \end{bmatrix} \in M_2(\mathbb{Z})$ such that $\det(A) = 1$ or $-1$.\\
On $M_2(\mathbb{R})$, $A$ is nonsingular since $\det(A) \neq 0$, so $A$ has an inverse matrix $A^{-1}$.\\
The inverse of $2\times2$ matrix is that
\[
A^{-1} = \frac{1}{\det(A)} 
\begin{bmatrix}
    d & -b \\
    -c & a
\end{bmatrix}
= \pm
\begin{bmatrix}
    d & -b \\
    -c & a
\end{bmatrix}
\]
Since $a, -b, -c, d \in \mathbb{Z}$, we obtain that $A^{-1} \in M_2(\mathbb{Z})$\\
Thus, $A$ is a unit in $M_2(\mathbb{Z})$.

\subsection*{$(\Rightarrow)$}
Suppose $A$ is a unit in $M_2(\mathbb{Z})$.\\
Then there exists $A^{-1} \in M_2(\mathbb{Z})$ such that $AA^{-1} = A^{-1}A = I_2$.\\
We derive that
\begin{align*}
    \det(AA^{-1}) &= \det(I_2)\\
    \det(A)\cdot \det(A^{-1}) &= 1
\end{align*}
Since $A, A^{-1} \in M_2(\mathbb{Z})$, we obtain that $\det(A), \det(A^{-1}) \in \mathbb{Z}$.\\
the only solution of $a\cdot b=1$ in $\mathbb{Z}$ is that $1\cdot1 = (-1)\cdot(-1) = 1$\\
Thus, $\det(A) = 1$ or $-1$.

\subsection*{$\therefore A \text{ is a unit in } M_2(\mathbb{Z}) \Longleftrightarrow  \det(A) = 1 \text{ or } -1$}
