\section*{Problem}
Let $R$ be a ring. Prove that the subset
\[
Z =\{z \in R \mid zr=rz \text{ for all } r \in R\}
\]
of $R$ is a subring of $R$.

\section*{Proof}
Let $R$ be a ring.\\
We will show the following to prove $Z$ is a subring of $R$.
\begin{enumerate}
    \item $0_R \in Z$
    \item $a+(-b) \in Z$ for all $a,b \in Z$
    \item $ab \in Z$ for all $a,b\in Z$
\end{enumerate}

\subsection*{1.}
On $R$, we derive that
\[
0_R = r \cdot 0_R = 0_R \cdot r \text{ for all } r \in R
\]
Thus $0_R \in Z$
