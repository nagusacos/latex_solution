\section*{Problem}
Let $R$ be a ring.
Prove that
\begin{itemize}
    \item The subset $Z = \{z \in R \mid zr =rz \text{ for all } r \in R\}$ is a subring of $R$.
\end{itemize}

\section*{Proof}
Let $R$ be a ring.\\
We will show the followings to prove $Z$ is a subring of $R$.
\begin{enumerate}
    \item $0_R \in Z$
    \item $a+(-b) \in Z$ for all $a,b \in Z$
    \item $ab \in Z$ for all $a,b\in Z$
\end{enumerate}
